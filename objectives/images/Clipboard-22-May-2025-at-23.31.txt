function markdown($text) {
    // Try to use Parsedown if available, else plain text with line breaks
    if (file_exists(__DIR__.'/Parsedown.php')) {
        require_once __DIR__.'/Parsedown.php';
        $Parsedown = new Parsedown();
        return $Parsedown->text($text);
    } else {
        return '<pre style="white-space: pre-wrap; word-break: break-word;">'.htmlspecialchars($text).'</pre>';
    }
}
